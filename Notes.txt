For Title Case file name and title metadata:
$TextInfo = (Get-Culture).TextInfo
Get-ChildItem *.ogg | ForEach-Object {
    $CleanName = $_.BaseName -replace '_', ' '
    
    $FinalName = $TextInfo.ToTitleCase($CleanName)
    
    ffmpeg -i $_.Name -metadata title="$FinalName" -c copy "$FinalName.ogg"
}
