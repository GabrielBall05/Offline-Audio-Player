For Title Case and title/name metadata as well:
$TextInfo = (Get-Culture).TextInfo
Get-ChildItem *.ogg | ForEach-Object {
    # 1. Remove extension and replace underscores with spaces
    $CleanName = $_.BaseName -replace '_', ' '
    
    # 2. Capitalize the first letter of every word (Title Case)
    $FinalName = $TextInfo.ToTitleCase($CleanName)
    
    # 3. Run ffmpeg with the new name as the Title metadata
    ffmpeg -i $_.Name -metadata artist="Amos Roddy" -metadata title="$FinalName" -c copy "$FinalName.ogg"
}
